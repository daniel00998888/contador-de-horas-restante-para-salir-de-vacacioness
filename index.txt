import random
from transformers import pipeline

# Cargar el modelo y el pipeline de conversación
chatbot = pipeline("conversational", model="microsoft/DialoGPT-medium")

# Definir algunas respuestas cordiales
greetings = ["¡Hola! Buenas, un gusto en volverte a ver. ¿En qué puedo ayudarte hoy?",
             "¡Hola de nuevo! ¿Cómo puedo asistirte hoy?",
             "¡Hola! Espero que estés teniendo un buen día. ¿En qué puedo ayudarte?"]

if __name__ == "__main__":
    print(random.choice(greetings))
    while True:
        user_input = input("Tú: ")
        if user_input.lower() in ['salir', 'adiós', 'chao']:
            print("¡Hasta luego! Fue un placer ayudarte.")
            break
        response = chatbot(user_input)
        print("IA: " + response[0]['generated_text'])
